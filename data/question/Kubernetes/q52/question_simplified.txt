Modify existing DaemonSet by adding Init Container using busybox image. Init Container creates /var/shared-data/hello.txt file with content "Hello from Init".
```
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: my-ds
spec:
  selector:
    matchLabels:
      name: my-ds
  template:
    metadata:
      labels:
        name: my-ds
    spec:
      containers:
      - name: nginx
        image: nginx
        volumeMounts:
        - name: shared-data
          mountPath: /var/shared-data
      volumes:
      - name: shared-data
        emptyDir: {}
```
