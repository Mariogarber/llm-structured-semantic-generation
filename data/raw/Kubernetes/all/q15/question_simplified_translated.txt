定义一个 DaemonSet，使用镜像 "quay.io/fluentd_elasticsearch/fluentd:v2.5.2" 来定义 "fluentd-elasticsearch" 容器。
```
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: fluentd-elasticsearch
spec:
  selector:
    matchLabels:
      name: fluentd-elasticsearch
  template:
    metadata:
      labels:
        name: fluentd-elasticsearch
    spec:
```
