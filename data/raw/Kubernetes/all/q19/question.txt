```

kind: DaemonSet
metadata:
  name: initcontainer
spec:
  selector:
    matchLabels:
      app: initcontainer
  template:
    metadata:
      labels:
        app: initcontainer
    spec:
      containers:
      - name: busybox
        image: busybox
        command: ["sleep", "3600"]
```
