Modify existing DaemonSet to:
- Set CPU limit to 500m and memory limit to 100Mi for container.
- Mount volume shared-data to /var/shared-data in container.
```
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: my-ds
spec:
  selector:
    matchLabels:
      name: my-ds
  template:
    metadata:
      labels:
        name: my-ds
    spec:
      nodeSelector:
        kubernetes.io/os: linux
      containers:
      - name: nginx
        image: nginx
        livenessProbe:
          httpGet:
            path: /
            port: 80
          initialDelaySeconds: 5
          timeoutSeconds: 1
        readinessProbe:
          httpGet:
            path: /
            port: 80
          initialDelaySeconds: 5
          timeoutSeconds: 1
```
