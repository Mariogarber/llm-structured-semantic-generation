Create a service with load balancer using the nginx selector, exposed on port 80. It should be accessible via browser. Use the following YAML:
```
apiVersion: apps/v1                                                                                                     
kind: Deployment                                                                                                        
metadata:                                                                                                               
  name: nginx-deployment                                                                                                
spec:                                                                                                                   
  replicas: 3                                                                                                           
  selector:                                                                                                             
    matchLabels:                                                                                                        
      app: nginx                                                                                                        
  template:                                                                                                             
    metadata:                                                                                                           
      labels:                                                                                                           
        app: nginx                                                                                                      
    spec:                                                                                                               
      containers:                                                                                                       
      - name: nginx-container                                                                                           
        image: nginx:latest                                                                                             
        ports:                                                                                                          
        - containerPort: 80
```
